**handleRefresh:**
```typescript
async function handleRefresh() {
  setRefreshing(true)
  try {
    const data = await window.api.sustentacao.refresh()
    setSnapshot(data)
  } catch (err) {
    setError(err instanceof Error ? err.message : 'Erro ao atualizar')
  } finally {
    setRefreshing(false)
  }
}
```
